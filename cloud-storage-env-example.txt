# Example environment variables for S3 cloud storage configuration
# Add these to your docker-compose.yml or .env file

# Enable cloud storage caching
ABS_CLOUD_STORAGE_ENABLED=true

# Storage provider: currently only 's3' is supported
ABS_CLOUD_STORAGE_PROVIDER=s3

# Installation:
# npm install @aws-sdk/client-s3 @aws-sdk/lib-storage

# Local cache path (used for all providers as temporary cache)
ABS_CLOUD_STORAGE_CACHE_PATH=/metadata/cache

# === S3 Configuration ===
ABS_S3_REGION=us-east-1
ABS_S3_BUCKET=audiobookshelf-cache
ABS_S3_ACCESS_KEY=your-access-key-here
ABS_S3_SECRET_KEY=your-secret-key-here
# Optional: for S3-compatible services like MinIO, DigitalOcean Spaces, etc.
# ABS_S3_ENDPOINT=https://nyc3.digitaloceanspaces.com
